208 def __permission(key):
209  try:
210  contentType = ContentType.objects.get_for_model(key[0])
211  permission = Permission.objects.filter(codename = key[1], content_type = contentType)
212  if permission:
213  return permission[0]
214  else:
215  raise DoesNotExist
216  except OperationalError:
217  return None
